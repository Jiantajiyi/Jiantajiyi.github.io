raid-廉价磁盘冗余阵列

发表于 2019-05-11 | 更新于: 2019-05-11 | 分类于 RHCE | 阅读次数:
字数统计: 1.7k | 阅读时长 ≈ 10

廉价磁盘冗余阵列（Redundant Array of inexpensive Disks）

准备11块新硬盘

[root@server_1 ~]# lsblk

raid0

[root@server_1 ~]# mdadm -C /dev/md0 -l 0 -n 2 /dev/sdb /dev/sdc

[root@server_1 ~]# mdadm -Ds

[root@server_1 ~]# mdadm -Ds > /etc/mdadm.conf
[root@server_1 ~]# mdadm -D /dev/md0

[root@server_1 ~]# mkfs.ext4 /dev/md0

[root@server_1 ~]# blkid /dev/md0

[root@server_1 ~]# mkdir /raid0
[root@server_1 ~]# vim /etc/fstab

[root@server_1 ~]# mount -a
[root@server_1 ~]# df -Th

raid1

[root@server_1 ~]# mdadm -C -v /dev/md1 -l 1 -n 2 -x 1 /dev/sdd /dev/sde /dev/sdf

[root@server_1 ~]# mdadm -D /dev/md1

[root@server_1 ~]# mdadm -Ds

[root@server_1 ~]# mdadm -Ds > /etc/mdadm.conf
[root@server_1 ~]# mkfs.ext4 /dev/md1

[root@server_1 ~]# mkdir /raid1
[root@server_1 ~]# mount /dev/md1 /raid1
[root@server_1 ~]# df -Th

raid5

[root@server_1 ~]# mdadm -C -v /dev/md5 -l 5 -n 3 -x 1 -c 32 /dev/sdg /dev/sdh /dev/sdi /dev/sdj

[root@server_1 ~]# mdadm -D /dev/md5

[root@server_1 ~]# mdadm -Dsv

[root@server_1 ~]# mdadm -Dsv > /etc/mdadm.conf
[root@server_1 ~]# mkdir /raid5
[root@server_1 ~]# mkfs.ext4 /dev/md5

[root@server_1 ~]# mount /dev/md5 /raid5
[root@server_1 ~]# df -Th

raid10

准备如下分区
[root@server_1 ~]# lsblk /dev/sdk

[root@server_1 ~]# mdadm -C -v /dev/md10 -l 10 -n 4 /dev/sdk1 /dev/sdk2 /dev/sdk3 /dev/sdk5

[root@server_1 ~]# cat /proc/mdstat

[root@server_1 ~]# mdadm -Dsv > /etc/mdadm.conf
[root@server_1 ~]# mkdir /raid10
[root@server_1 ~]# mkfs.ext4 /dev/md10

[root@server_1 ~]# mount /dev/md10 /raid10/
[root@server_1 ~]# df -Th

删除磁盘中所有raid信息

[root@server_1 ~]# umount /raid*
[root@server_1 ~]# mdadm -Ss
[root@server_1 ~]# rm -rf /etc/mdadm.conf
[root@server_1 ~]# mdadm --zero-superblock /dev/sd{b,c,d,e,f,g,h,i,j,k}
删除之前添加的开机自动挂载的项
[root@server_1 ~]# vim /etc/fstab
